# -------------------------
# Register Initialization
# -------------------------
addi x1, x0, 15          # x1 = 15
addi x2, x0, 10          # x2 = 10

# -------------------------
# Arithmetic
# -------------------------
add  x3, x1, x2          # x3 = 25
sub  x4, x1, x2          # x4 = 5

# -------------------------
# Logic
# -------------------------
and  x5, x1, x2          # x5 = 10
or   x6, x1, x2          # x6 = 15
xor  x7, x1, x2          # x7 = 5

# -------------------------
# Shift
# -------------------------
addi x8, x0, 2           # shift amount
sll  x9, x2, x8          # x9 = 40
srl  x10, x9, x8         # x10 = 10

# -------------------------
# Compare
# -------------------------
slt  x11, x2, x1         # x11 = 1
sltu x12, x2, x1         # x12 = 1

# -------------------------
# Memory
# -------------------------
addi x13, x0, 32         # address
sw   x3,0(x13)           # MEM[32]=25
lw   x14,0(x13)          # x14=25

# -------------------------
# Branch
# -------------------------
beq  x3,x14,SKIP

addi x15,x0,99           # should never execute

SKIP:

addi x16,x0,55

nop
